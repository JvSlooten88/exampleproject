[docs]    def demean(self, standardize=True):
        """ Demeans array and optionally standardizes it.

        Parameters
        ----------
        standardize : bool
            Whether to standardize (divide by std).

        Returns
        -------
        arr_out : array-like
            Demeand array.

        """

        arr_out = demean(self.arr, to_file=False)

        if standardize:
            arr_out = arr_out / arr_out.std(axis=0)

        return arr_out
